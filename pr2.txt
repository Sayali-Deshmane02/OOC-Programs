#include <iostream>
using namespace std;
class Complex
{
    public:
            int real,imag;
    public:
            Complex()
            {
                real=0;
                imag=0; 
            }
            Complex(int r,int i)
            {
                real =r;
                imag =i;
            }
            Complex addComplexNumber(Complex c1,Complex c2)
            {
                Complex res;
                res.real=c1.real+c2.real;
                res.imag=c1.imag+c2.imag;
                return res;
            }
           
};
int main()
{
    Complex c1(4,5);
    cout<<"First Complex Nmber:"<<c1.real<<"+"<<c1.imag<<"i";
    Complex c2(6,8);
    cout<<"\nSecond Complex Nmber:"<<c2.real<<"+"<<c2.imag<<"i";
    Complex c3;
    c3=c3.addComplexNumber(c1,c2);
    cout<<"\nSum of complex numbers is:"<<c3.real<<"+"<<c3.imag<<"i";
    return 0;
}

Output:
First Complex Nmber:4+5i
Second Complex Nmber:6+8i
Sum of complex numbers is:10+13i
